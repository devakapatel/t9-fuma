```java twoslash lineNumbers title="Java"
// Student.java
class Student {
    private String name;
    private int age;
    private String id;
    private double gpa;

    // Constructor
    public Student(String name, int age, String id, double gpa) {
        this.name = name;
        this.age = age;
        this.id = id;
        this.gpa = gpa;
    }

    // Getter methods
    public String getName() {
        return name;
    }

    public int getAge() {
        return age;
    }

    public String getId() {
        return id;
    }

    public double getGpa() {
        return gpa;
    }

    // Update GPA
    public void updateGpa(double newGpa) {
        this.gpa = newGpa;
    }

    // Display student details
    public void displayInfo() {
        System.out.println("Student Name: " + name);
        System.out.println("Age: " + age);
        System.out.println("ID: " + id);
        System.out.println("GPA: " + gpa);
        System.out.println("--------------------");
    }
}

// Main.java
public class Main {
    public static void main(String[] args) {
        Student s1 = new Student("Amit", 20, "ST101", 3.6);
        Student s2 = new Student("Priya", 21, "ST102", 3.9);

        s1.displayInfo();
        s2.displayInfo();

        s1.updateGpa(3.8);
        System.out.println("After GPA update for "+ s1.getName() + ":");
        s1.displayInfo();
    }
}
```
